    function savedata() {
        const password = document.getElementById("password").value;
        const email = document.getElementById("email").value;
    
        const user_records = JSON.parse(localStorage.getItem("users")) || [];
    
        const userFound = user_records.some(v => v.email === email && v.password === password);
    
        if (userFound) {
            alert("Login Successful");
    
            const currentUser = user_records.find(v => v.email === email && v.password === password);
    
            localStorage.setItem("username", currentUser.username);
            localStorage.setItem("email", currentUser.email);
            localStorage.setItem("phone_number", currentUser.phone_number);
            localStorage.setItem("birth_date", currentUser.birth_date);
    
            window.location.href = "profile.html";
        } else {
            alert("Login failed");
        }
    }

  function savedata()
{
let username,password,email,phone_number,birth_date;
username=document.getElementById("username").value;
password=document.getElementById("password").value;
email=document.getElementById("email").value;
phone_number=document.getElementById("phone_number").value;
birth_date=document.getElementById("birth_date").value;

let user_records=new Array();
user_records=JSON.parse(localStorage.getItem("users"))?JSON.parse(localStorage.getItem("users")):[]
if(user_records.some((v)=>{return v.email==email}))
{
  alert("duplicate data");
}
else
{
  user_records.push({
  "username":username,
  "password":password,
  "email":email,
  "phone_number":phone_number,
  "birth_date":birth_date,
 
})
localStorage.setItem("users",JSON.stringify(user_records));
}

}

function logout(){
    localStorage.removeItem("username");
    localStorage.removeItem("email");
    localStorage.removeItem("password");
    localStorage.removeItem("phone_number");
    localStorage.removeItem("birth_date");
    window.location.href = "login.html";
    
}
